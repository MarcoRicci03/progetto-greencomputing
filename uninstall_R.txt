# Istruzioni per disinstallare R su Ubuntu Server
# (dopo aver completato il progetto Green Computing)

## 1. Disinstalla R base e tutti i pacchetti
sudo apt-get remove --purge r-base r-base-core r-base-dev r-recommended -y
sudo apt-get autoremove --purge -y

## 2. Rimuovi eventuali pacchetti R aggiuntivi installati
sudo apt-get remove --purge "r-cran-*" -y
sudo apt-get autoremove --purge -y

## 3. Rimuovi le librerie R installate dall'utente (pacchetti R installati con install.packages())
# La cartella di default è ~/R/
rm -rf ~/R/

## 4. Rimuovi eventuali file di configurazione residui
rm -f ~/.Rhistory
rm -f ~/.Rprofile
rm -rf ~/.R/

## 5. (Opzionale) Se hai aggiunto il repository CRAN manualmente, rimuovilo
# Controlla se esiste:
ls /etc/apt/sources.list.d/ | grep -i cran
# Se trovi un file tipo "cran.list", rimuovilo:
# sudo rm /etc/apt/sources.list.d/cran.list
# Poi aggiorna:
# sudo apt-get update

## 6. Verifica che R sia stato rimosso correttamente
which Rscript   # deve restituire nulla
R --version     # deve restituire "command not found"

## NOTE:
# - Questi comandi rimuovono R e tutti i pacchetti CRAN installati via apt.
# - Se hai installato pacchetti direttamente con install.packages() in R,
#   essi si trovano in ~/R/x86_64-pc-linux-gnu-library/ e vengono rimossi
#   dal comando al punto 3.
# - Il venv Python (cartella venv/) e i pacchetti Python NON vengono toccati.
